module test;
using test::*;
using system::*;

struct Value
{
	value : int;
}

func ReadValue(value : object) : string
{
	var nullable = cast (Value?) value;
	return nullable is null ? "null" : nullable.value;
}

func main():string
{
	return ReadValue(null) & ", " & ReadValue({value : 0} of Value);
}
